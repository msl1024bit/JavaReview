##排序算法
###插入排序(O(N?))
```
    public class InsertSort {
        private static  void sort(int[] array) {
            for (int i = 1; i < array.length; i++) {
                int temp = array[i];
                for (int j = i; j > 0 && temp < array[j - 1]; j--) {
                    array[j] = array[j - 1];
                    array[j - 1] = temp;
                }
            }
        }
    
        public static void main(String[] args) {
            int[] array = {10,7,8,1,2,200,178,6};
            sort(array);
            for (int i = 0; i < array.length; i++) {
                System.out.println(array[i]);
            }
        }
    }
```

###快速排序
```
    public class QuickSort {

        public static void main(String[] args) {
            int[] arr = {2,4,1,0,10,213,22,11,90,29,4};
            sort(arr, 0, arr.length -1);
            for (int i = 0; i < arr.length; i++) {
                System.out.println(arr[i]);
            }
        }
    
        private static void sort(int[] arr, int start, int end) {
            if (start > end) {
                return;
            }
            int pivot = partition(arr, start, end);
            sort(arr, start, pivot - 1);
            sort(arr, pivot + 1, end);
    
        }
    
        /**
         * 分区
         *
         * @param arr
         * @param start
         * @param end
         * @return
         */
        private static int partition(int[] arr, int start, int end) {
            // 首先定义左边第一个数为基准书
            int pivot = start;
            int i = start;
            int j = end;
            while (i < j) {
                // 从右边开始遍历直到找到小于基准数的数
                while (arr[j] >= arr[pivot] && i < j) {
                    j--;
                }
    
                // 从左边开始遍历直到找到大于基准数的数
                while (arr[i] < arr[pivot] && i < j) {
                    i++;
                }
    
                // 如果这时i和j没有碰上就交换两者位置
                if (i < j) {
                    swap(arr, i, j);
                }
            }
            // 如果这时i和j碰上就把基准数换到i和j碰面的位置
            swap(arr, pivot, j);
            pivot = j;
            return pivot;
        }
    
        /**
         * 交换两个数位置
         *
         * @param arr
         * @param a
         * @param b
         */
        private static void swap(int[] arr, int a, int b) {
            int temp = arr[a];
            arr[a] = arr[b];
            arr[b] = temp;
        }
    }
```

###快排（单链表实现）
```
    public class LinkedListQuickSort {
        public static void main(String[] args) {
            int[] arr = {2,4,1,0,10,213,22,11,90,29,4};
            LinkedList<Integer> list = new LinkedList<>();
            for (int i = 0; i < arr.length; i++) {
                list.add(arr[i]);
            }
            sort(list, 0, arr.length -1);
    
            for (int i = 0; i < list.size(); i++) {
                System.out.println(list.get(i));
            }
        }
    
        private static void swap(LinkedList<Integer> list, int a, int b) {
            Integer temp = list.get(a);
            list.set(a, list.get(b));
            list.set(b, temp);
        }
    
        private static void sort(LinkedList<Integer> list, int start, int end) {
            if (start > end) {
                return;
            }
    
            // 左边第一个数作为基准数
            int pivot = start;
    
            int i = start;
            int j = i + 1;
    
            // 大循环条件，j不能超过链表长度
            while (j <= end) {
                // 如果 j 指向的值大于等于基准数字（如果比基准大，直接跳过）
                while (j <= end && list.get(j) >= list.get(pivot)) {
                    j++;
                }
    
                // 否则，j 指向的值小于基准，则交换
                if (j <= end) {
                    i++;
                    swap(list, i, j);
                    j++;
                }
            }
            swap(list, pivot, i);
            pivot = i;
    
            sort(list, start, pivot - 1);
            sort(list, pivot + 1, end);
    
        }
    }

```
